40 std::string LcioFileNamer::getFieldValue(std::string field) {
41  std::string value;
42 
43  static std::string sep("-");
44 
45  if (field == "application") {
46  value = PackageInfo::getShortName() + "-v" + PackageInfo::getVersion(sep);
47  } else if (field == "geometry") {
48  value = LCDDProcessor::instance()->getDetectorName();
49  } else if (field == "date") {
50  value = TimeUtil::getDate();
51  } else if (field == "event" || field == "evt") {
52  value = LcioFileNamer::makeEventName();
53  } else if (field == "eventNumber") {
54  value = LcioFileNamer::makeEventNumberString();
55  } else if (field == "run") {
56  value = LcioFileNamer::makeRunNumberString();
57  } else if (field == "binary") {
58  value = SlicApplication::instance()->getBinaryBasename();
59  } else if (field == "physics") {
60  value = PhysicsListManager::instance()->getCurrentListName();
61  } else {
62 #ifdef SLIC_LOG
63  log() << LOG::error << "Ignoring unknown autoname field <" << field << ">." << LOG::done;
64 #endif
65  value = "";
66  }
67  return value;
