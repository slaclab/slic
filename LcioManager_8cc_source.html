81 void LcioManager::openLcioFile() {
82 
83  // Get full output path with extension to check its existence.
84  string fullFilename = getFullOutputPath(true);
85 
86  // Default to writing a new file.
87  int writeMode = LCIO::WRITE_NEW;
88 
89  // File exists?
90  if (FileUtil::fileExists(fullFilename)) {
91 
92  // Failure mode is on, so kill the current run.
93  if (m_fileExistsAction == eFail) {
94  log() << LOG::fatal << LOG::name << LOG::sep << "LCIO file already exists: " << fullFilename << LOG::done;
95  RunManager::instance()->abortRun(SlicApplication::OUTPUT_FILE_EXISTS);
96  // Deletion mode is on, so try to remove the current file.
97  } else if (m_fileExistsAction == eDelete) {
98  if (FileUtil::removeFile(fullFilename) != 0) {
99  // Fatal error. File could not be removed.
100  log() << LOG::fatal << LOG::head << "Unable to delete old LCIO file: " << fullFilename << LOG::done;
101  RunManager::instance()->abortRun(SlicApplication::FAILED_DELETE_LCIO_FILE);
102  } else {
103  log().okay("Deleted old LCIO file: " + fullFilename);
104  }
105  }
106  // Append mode is on.
107  else if (m_fileExistsAction == eAppend) {
108  log().okay("Appending to existing LCIO file: " + fullFilename);
109  writeMode = LCIO::WRITE_APPEND;
110  }
111  }
112  // Create a new file if doesn't exist.
113  else {
114  log().okay("Creating new LCIO file: " + fullFilename);
115  }
116 
117  // Open the file using the writer with the full path.
118  if (m_writer) {
119  if (!RunManager::instance()->isRunAborted()) {
120  m_writer->open(getFullOutputPath(false), writeMode);
121  m_writerIsOpen = true;
122  }
123  } else {
124  //G4Exception("", "", FatalException, "LCWriter is null.");
125  log() << LOG::fatal << LOG::head << "LCWriter is null." << LOG::done;
126  RunManager::instance()->abortRun();
127  }
128 }
