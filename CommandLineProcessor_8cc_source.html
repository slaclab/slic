30 void CommandLineProcessor::process(int argc, char** argv) {
31 
32  // Description of SLIC's options in getopt's format.
33  static struct option long_options[] = {
34  { "help", no_argument, 0, 'h' },
35  { "help", no_argument, 0, '?' },
36  { "version", no_argument, 0, 'v' },
37  { "interactive", no_argument, 0, 'n' },
38  { "macro", required_argument, 0, 'm' },
39  { "event-file", required_argument, 0, 'i' },
40  { "lcdd-url", required_argument, 0, 'g' },
41  { "lcio-file", required_argument, 0, 'o' },
42  { "lcio-path", required_argument, 0, 'p' },
43  { "autoname", optional_argument, 0, 'O' },
44  { "lcio-delete", no_argument, 0, 'x' },
45  { "run-events", required_argument, 0, 'r' },
46  { "skip-events", required_argument, 0, 's' },
47  { "physics-list", required_argument, 0, 'l' },
48  { "log-file", required_argument, 0, 'L' },
49  { "seed", required_argument, 0, 'd' },
50  { "dump-gdml", required_argument, 0, 'G' },
51  { "optical", no_argument, 0, 'c' },
52  { "pdg-file", required_argument, 0, 'P' },
53  { "lcdd-version", required_argument, 0, 'V' },
54  { "lcdd-setup", required_argument, 0, 'S' },
55  { "quiet", no_argument, 0, 'q' },
56  { 0, 0, 0, 0 }
57  };
58 
59  const char* optstr = m_getoptOptions.c_str();
60 
61  while (1) {
62  int option_index = 0;
63  int c;
64 
65  // Call getopt for next option.
66  c = getopt_long(argc, argv, optstr, long_options, &option_index);
67 
68 #ifdef SLIC_LOG
69  log() << LOG::verbose << "Handling option <" << c << ">." << LOG::done;
70 #endif
71 
72  // Done.
73  if (c == -1)
74  break;
75 
76  // Handle options.
77  switch (c) {
78 
79  case 'h':
80  m_help_flag = 1;
81  break;
82 
83  case '?':
84  m_help_flag = 1;
85  break;
86 
87  case 'n':
88  m_interactive_flag = 1;
89  break;
90 
91  case 'v':
92  m_version_flag = 1;
93  break;
94 
95  default:
96 
97  stringstream ostr;
98  char cc = (char) c;
99  ostr << cc;
100  string theopt = ostr.str();
101  if (optarg) {
102  m_commandline.push_back(OptionsPair(theopt, optarg));
103  } else {
104  m_commandline.push_back(OptionsPair(theopt, ""));
105  }
106 
107  break;
108  }
109  }
110 
111  //
112  // The processing flow...
113  //
114  // 1) If version flag is set, print version and exit.
115  //
116  // 2) If usage flag is set or no arguments were received,
117  // print the usage and exit.
118  //
119  // 3) If a single non-option command was received, treat
120  // it as a macro to execute.
121  //
122  // 4) If no options were received or the number of options
123  // is less than the number of arguments, print usage
124  // and exit (syntax incorrect).
125  //
126  // 5) Finally, process the options in order if the syntax
127  // appears to be okay.
128  //
129 
130  // Print the version information and abort.
131  if (m_version_flag) {
132  printVersion();
133  abort();
134  }
135  // Print the usage and abort if the help flag is set
136  // or no arguments were received.
137  else if (m_help_flag || argc == 1) {
138  printUsage();
139  abort();
140  }
141  // Single macro to execute, if got 1 argument that is a non-option.
142  else if (argc == 2 && optind < argc) {
143  stringstream str;
144  str << "/control/execute ";
145  str << argv[1];
146  m_g4q.add(str.str());
147  }
148  // Probably extra arguments or no options.
149  else if (optind == 1 || optind < argc) {
150  printUsage();
151  abort();
152  }
153  // Default is process all the options and build the CommandQueue.
154  else {
155  processOptions();
156  }
157 }

262 void CommandLineProcessor::processOption(const string& opt) {
263 
264  // Loop over all input arguments to look for this option.
265  for (CommandLineArguments::iterator it = m_commandline.begin(); it != m_commandline.end(); it++) {
266 
267 
268  // Look for a matching switch in the arguments.
269  if (it->first == opt) {
270 
271  // Get the associated specification for this option.
272  CommandLineOption* cmdlineopt = getCommandLineOption(opt);
273 
274  // We must have a valid option!
275  assert(cmdlineopt != 0);
276 
277  // Get the associated Geant4 command string.
278  string cmdstr = cmdlineopt->getG4CommandString();
279 
280  // Add arguments to the command string, if some were passed.
281  if (it->second != "") {
282  cmdstr += " ";
283  cmdstr += it->second;
284  }
285 
286  // Add the command string to the CommandQueue.
287  m_g4q.add(cmdstr);
288  }
289  }
290 }
