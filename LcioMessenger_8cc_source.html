38  // set output path

46  // fileExistsAction

49  }

53  }
54  // autoname
55  else if (cmd == m_autonameCmd) {
56  std::vector<std::string> fields;
57  StringUtil::split(newVals, " ", fields);
58  m_mgr->setAutonameFields(fields);
59  }
60  /* dump event */
61  else if (cmd == m_printEventCmd) {
62  m_mgr->enableDumpEvent(m_printEventCmd->GetNewBoolValue(newVals.c_str()));
63  }
64  // flags
65  else if (cmd == m_setLongFlagCmd || cmd == m_setPDGFlagCmd) {
66 
67  // no args then true
68  bool flag_set = true;
69 
70  // get arg val if exists
71  if (s != std::string("")) {
72  flag_set = StringUtil::toBool(s);
73  }
74 
75  LcioHitsCollectionBuilder* hcb = m_mgr->getHitsCollectionBuilder();
76 
77  if (hcb) {
78 
79  // CHBIT_LONG for positions in hits
80  if (cmd == m_setLongFlagCmd) {
81  hcb->setLongFlag(flag_set);
82  }
83  // CHBIT_PDG for every PDG contribution in hits
84  else if (cmd == m_setPDGFlagCmd) {
85  hcb->setPDGFlag(flag_set);
86  }
87  } else {
88  G4Exception("", "", FatalException, "LcioHitsCollectionBuilder is null.");
89  }
90  } else if (cmd == _writeCompleteEventCmd) {
91  G4bool writeCompleteEvent = StringUtil::toBool(s);
92  LcioManager::instance()->setWriteCompleteEvent(writeCompleteEvent);
93  }
94  // bad command; shouldn't happen
95  else {
96  G4Exception("LcioMessenger::setNewValue()", "", JustWarning, "Unknown command.");
97  }
98  return;
