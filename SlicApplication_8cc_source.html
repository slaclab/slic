174 void SlicApplication::run() {
175  // Initialize if not already.
176  if (!m_isInitialized) {
177  initialize();
178  }
179 
180  // Retrieve the list of Geant4 commands created by CommandLineProcessor.
181  CommandQueue* q = CommandLineProcessor::instance()->getCommandQueue();
182 
183  // Print, execute, and clear the CommandQueue.
184 #ifdef SLIC_LOG
185  log().verbose("Printing the Geant4 command queue ...");
186  q->printOut();
187 #endif
188  q->execute();
189  q->clear();
190 
191  // Start the UI session if in interactive mode.
192  if (getMode() == eInteractive) {
193  log().verbose("Starting interactive session ...");
194  m_ui->SessionStart();
195  delete m_ui;
196  }
197 }
